🎖️ Advanced Function Concepts

Scope and Closures

# Local vs Global scope
global_var = "I'm global"

def outer_function():
    outer_var = "I'm in outer"
    
    def inner_function():
        inner_var = "I'm in inner"
        print(f"Access: {global_var}, {outer_var}, {inner_var}")
    
    return inner_function

# Closure: inner function remembers outer scope
closure = outer_function()
closure()  # Access: I'm global, I'm in outer, I'm in inner

# Practical closure example: function factory
def create_multiplier(factor):
    def multiplier(x):
        return x * factor
    return multiplier

double = create_multiplier(2)
triple = create_multiplier(3)

print(double(5))  # 10
print(triple(5))  # 15

⚡ Expert Insight: Closures are powerful for creating specialized functions, decorators, and maintaining state without classes.

⚡ Generators

# Generator function
def count_up_to(max):
    count = 1
    while count <= max:
        yield count
        count += 1

# Use generator
counter = count_up_to(3)
for num in counter:
    print(num)  # 1, 2, 3

# Generator expression
squares_gen = (x**2 for x in range(1000000))  # Memory efficient!

# Infinite generator
def fibonacci():
    a, b = 0, 1
    while True:
        yield a
        a, b = b, a + b

Generators are memory-efficient because they produce values on-demand instead of storing everything in memory!

🔒 Encapsulation: Data Protection

Access Control

class BankAccount:
    def __init__(self, account_number, initial_balance):
        self.account_number = account_number
        self._balance = initial_balance        # Protected (convention)
        self.__pin = "1234"                   # Private (name mangling)
    
    def deposit(self, amount):
        if amount > 0:
            self._balance += amount
            return f"Deposited ${amount}. New balance: ${self._balance}"
        return "Invalid deposit amount"
    
    def withdraw(self, amount, pin):
        if pin != self.__pin:
            return "Invalid PIN"
        if amount > self._balance:
            return "Insufficient funds"
        if amount > 0:
            self._balance -= amount
            return f"Withdrew ${amount}. New balance: ${self._balance}"
        return "Invalid withdrawal amount"
    
    def get_balance(self, pin):
        if pin != self.__pin:
            return "Invalid PIN"
        return f"Current balance: ${self._balance}"

account = BankAccount("12345", 1000)
print(account.deposit(500))           # Deposited $500. New balance: $1500
print(account.withdraw(200, "1234"))  # Withdrew $200. New balance: $1300

Properties: Pythonic Getters/Setters

class Temperature:
    def __init__(self, celsius=0):
        self._celsius = celsius
    
    @property
    def celsius(self):
        return self._celsius
    
    @celsius.setter
    def celsius(self, value):
        if value < -273.15:
            raise ValueError("Temperature below absolute zero is not possible")
        self._celsius = value
    
    @property
    def fahrenheit(self):
        return (self._celsius * 9/5) + 32
    
    @fahrenheit.setter
    def fahrenheit(self, value):
        self.celsius = (value - 32) * 5/9
    
    @property
    def kelvin(self):
        return self._celsius + 273.15

# Usage
temp = Temperature(25)
print(f"Celsius: {temp.celsius}")      # 25
print(f"Fahrenheit: {temp.fahrenheit}") # 77.0
print(f"Kelvin: {temp.kelvin}")        # 298.15

temp.fahrenheit = 100  # Set via Fahrenheit
print(f"Celsius: {temp.celsius}")      # 37.77777777777778

💡 Python Convention: Single underscore (_) = "internal use", double underscore (__) = "really private" (name mangled).

🎯 Advanced OOP Concepts

Magic Methods (Dunder Methods)

class Vector:
    def __init__(self, x, y):
        self.x = x
        self.y = y
    
    def __str__(self):
        return f"Vector({self.x}, {self.y})"
    
    def __repr__(self):
        return f"Vector(x={self.x}, y={self.y})"
    
    def __add__(self, other):
        return Vector(self.x + other.x, self.y + other.y)
    
    def __mul__(self, scalar):
        return Vector(self.x * scalar, self.y * scalar)
    
    def __eq__(self, other):
        return self.x == other.x and self.y == other.y
    
    def __len__(self):
        return int((self.x**2 + self.y**2)**0.5)
    
    def __getitem__(self, key):
        if key == 0:
            return self.x
        elif key == 1:
            return self.y
        else:
            raise IndexError("Vector index out of range")

# Using magic methods
v1 = Vector(3, 4)
v2 = Vector(1, 2)

print(v1)           # Vector(3, 4)
print(v1 + v2)      # Vector(4, 6)
print(v1 * 2)       # Vector(6, 8)
print(len(v1))      # 5
print(v1[0])        # 3 (accessing like a list)
print(v1 == v2)     # False

Class Methods & Static Methods

class Person:
    population = 0  # Class variable
    
    def __init__(self, name, age):
        self.name = name
        self.age = age
        Person.population += 1
    
    @classmethod
    def get_population(cls):
        return f"Total population: {cls.population}"
    
    @classmethod
    def from_string(cls, person_str):
        name, age = person_str.split('-')
        return cls(name, int(age))
    
    @staticmethod
    def is_adult(age):
        return age >= 18
    
    def __del__(self):
        Person.population -= 1

# Usage
p1 = Person("Alice", 25)
p2 = Person.from_string("Bob-30")  # Alternative constructor

print(Person.get_population())     # Total population: 2
print(Person.is_adult(16))         # False

from abc import ABC, abstractmethod

class Shape(ABC):
    @abstractmethod
    def area(self):
        pass
    
    @abstractmethod
    def perimeter(self):
        pass
    
    def describe(self):
        return f"This shape has area {self.area()}"

class Rectangle(Shape):
    def __init__(self, width, height):
        self.width = width
        self.height = height
    
    def area(self):
        return self.width * self.height
    
    def perimeter(self):
        return 2 * (self.width + self.height)

# Cannot instantiate abstract class
# shape = Shape()  # TypeError!

rect = Rectangle(5, 3)
print(rect.area())        # 15
print(rect.describe())    # This shape has area 15

🏗️ Design Patterns in Python

Singleton Pattern

class DatabaseConnection:
    _instance = None
    _initialized = False
    
    def __new__(cls):
        if cls._instance is None:
            cls._instance = super().__new__(cls)
        return cls._instance
    
    def __init__(self):
        if not self._initialized:
            self.connection_string = "localhost:5432"
            self._initialized = True
    
    def query(self, sql):
        return f"Executing: {sql}"

# Usage
db1 = DatabaseConnection()
db2 = DatabaseConnection()
print(db1 is db2)  # True - same instance

Factory Pattern

class AnimalFactory:
    @staticmethod
    def create_animal(animal_type, name):
        if animal_type.lower() == "dog":
            return Dog(name, 0, "Mixed")
        elif animal_type.lower() == "cat":
            return Cat(name, 0)
        else:
            raise ValueError(f"Unknown animal: {animal_type}")

# Usage
pet = AnimalFactory.create_animal("dog", "Buddy")
print(pet.speak())  # Buddy barks: Woof!

Observer Pattern

class Observable:
    def __init__(self):
        self._observers = []
    
    def attach(self, observer):
        self._observers.append(observer)
    
    def notify(self, message):
        for observer in self._observers:
            observer.update(message)

class EmailNotifier:
    def update(self, message):
        print(f"📧 Email: {message}")

class SMSNotifier:
    def update(self, message):
        print(f"📱 SMS: {message}")

class NewsService(Observable):
    def publish_news(self, headline):
        print(f"📰 Publishing: {headline}")
        self.notify(headline)

# Usage
news = NewsService()
news.attach(EmailNotifier())
news.attach(SMSNotifier())

news.publish_news("Python 4.0 Released!")
# 📰 Publishing: Python 4.0 Released!
# 📧 Email: Python 4.0 Released!
# 📱 SMS: Python 4.0 Released!

⚡ Expert Advice: Design patterns solve common problems, but don't overuse them. Python's simplicity often provides better solutions than complex patterns.

⚙️ Chapter 7: Python Internals

The Global Interpreter Lock (GIL)

What is the GIL? A mutex that protects access to Python objects, preventing multiple threads from executing Python code simultaneously.

GIL Impact:

  • Single-threaded execution for CPU-bound tasks
  • No performance gain from threading for compute
  • I/O operations release the GIL
  • Multiprocessing bypasses GIL limitations

Workarounds:

  • Use multiprocessing for CPU-bound tasks
  • Use asyncio for I/O-bound tasks
  • Use C extensions (NumPy, etc.)
  • Consider alternative implementations (PyPy)
